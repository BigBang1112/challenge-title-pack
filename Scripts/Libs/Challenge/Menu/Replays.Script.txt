Void Init() {
	ReplayBrowser::Init();
}

Void Start() {
	ReplayBrowser::Start();
}

Void Refresh() {
	ReplayBrowser::Refresh();
}

Void Event(CManiaAppEvent _Event) {
	declare Event <=> _Event;

	ReplayBrowser::Event(Event);

	if(Event.Type == CManiaAppEvent::EType::LayerCustomEvent) {
		if(Event.CustomEventLayer == Layers::Get("ReplayInfo"))
		switch(Event.CustomEventType) {
			case "Back": {
				Layers::Hide("Replays", True);
				Layers::Hide("ReplayInfo", True);
				Layers::Show("Menu", True);
				Layers::Show("Quickplay", True);
				Layers::Show("Top", True);
				Layers::Show("Playsessions", True);
				Layers::Show("News", True);
			}
			case "Open": {
				if(TitleControl.IsReady)
					TitleControl.ViewReplay(ReplayBrowser::GetSelectedElement().File);
			}
			case "Play": {
				if(TitleControl.IsReady) {
					Loading::SetLoading(ReplayBrowser::GetSelectedMap().CollectionName, ReplayBrowser::GetSelectedMap().MapUid);
					TitleControl.PlayMap(ReplayBrowser::GetSelectedMap().FileName, "", "");
				}
			}
		}
	}
}

Void Async() {
	ReplayBrowser::Async();
}

Void Loop() {
	ReplayBrowser::Loop();
}
